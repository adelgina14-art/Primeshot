export default async function handler(req, res) { // 1. التحقق من طريقة الطلب if (req.method !== 'POST') { return res.status(405).json({ error: 'Method not allowed' }); } const { prompt } = req.body; const apiKey = process.env.STABILITY_API_KEY; // 2. التحقق من وجود المفتاح في السيرفر if (!apiKey) { return res.status(500).json({ error: 'API Key is missing in Vercel settings' }); } try { const response = await fetch( 'https://api.stability.ai/v1/generation/stable-diffusion-v1-6/text-to-image', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Accept': 'application/json', 'Authorization': `Bearer ${apiKey.trim()}`, // تنظيف المفتاح من أي مسافات }, body: JSON.stringify({ text_prompts: [{ text: prompt }], cfg_scale: 7, height: 512, width: 512, steps: 30, samples: 1, }), } ); const data = await response.json(); if (!response.ok) { console.error('Stability Error:', data); return res.status(response.status).json({ error: data.message || 'فشل التوليد من المصدر' }); } // 3. إرسال الصورة بنجاح res.status(200).json({ image: data.artifacts[0].base64 }); } catch (error) { console.error('Server Error:', error.message); res.status(500).json({ error: 'حدث خطأ داخلي: ' + error.message }); } }
