export default async function handler(req, res) { // السماح فقط بطلبات POST if (req.method !== "POST") { return res.status(405).json({ error: "Method not allowed" }); } try { const { prompt } = req.body; if (!prompt) { return res.status(400).json({ error: "الرجاء كتابة وصف للصورة" }); } // ملاحظة: غيرنا الرابط للإصدار v1 لأنه الأكثر استقراراً مع JSON const response = await fetch( "https://api.stability.ai/v1/generation/stable-diffusion-v1-6/text-to-image", { method: "POST", headers: { "Content-Type": "application/json", "Accept": "application/json", "Authorization": `Bearer ${process.env.STABILITY_API_KEY.trim()}`, }, body: JSON.stringify({ text_prompts: [{ text: prompt }], cfg_scale: 7, height: 512, width: 512, steps: 30, samples: 1, }), } ); const result = await response.json(); if (!response.ok) { console.error("Stability API Error:", result); return res.status(response.status).json({ error: result.message || "فشل التوليد من المصدر" }); } // إرسال الصورة بصيغة Base64 إلى الموقع res.status(200).json({ image: result.artifacts[0].base64 }); } catch (error) { console.error("Server Error:", error.message); res.status(500).json({ error: "حدث خطأ في السيرفر: " + error.message }); } }
